1 package praktikum1;
 2 
 3 class Queue {
 4     int[] data;
 5     int front;
 6     int rear;
 7     int size;
 8     int max;
 9 
10     Queue(int n){
11         this.max = n;
12         this.data = new int[this.max];
13         this.size = 0;
14         this.front = this.rear = -1;
15     }
16 
17     boolean isEmpty(){
18         return this.size == 0;
19     }
20 
21     boolean isFull(){
22         return this.size == this.max;
23     }
24 
25     void enqueue(int dt){
26         if (this.isFull())
27             System.out.println("Queue sudah penuh");
28         else {
29             if (this.isEmpty())
30                 this.front = this.rear = 0;
31             else {
32                 if (this.rear == this.max-1)
33                     this.rear = 0;
34                 else
35                     this.rear++;
36             }
37             this.data[this.rear] = dt;
38             this.size++;
39         }
40     }
41 
42     int dequeue(){
43         int dt = 0;
44         if (this.isEmpty())
45             System.out.println("Queue masih kosong");
46         else {
47             dt = this.data[this.front];
48             this.size++;
49             if (this.isEmpty())
50                 this.front = this.rear = -1;
51             else {
52                 if (this.front == this.max-1)
53                     this.front = 0;
54                 else
55                     this.front++;
56             }
57         }
58         return dt;
59     }
60 
61     void peek(){
62         if (!this.isEmpty())
63             System.out.println("Elemen terdepan: " + this.data[this.front]);
64         else
65             System.out.println("Queue masih kosong");
66     }
67 
68     void print(){
69         if (!this.isEmpty())
70             System.out.println("Queue masih kosong");
71         else {
72             int i = front;
73             while (i != rear) {
74                 System.out.print(this.data[i] + " ");
75                 i = (i + 1) % this.max;
76             }
77             System.out.println(this.data[i] + " ");
78             System.out.println("Jumlah elemen = " + this.size);
79         }
80     }
81 
82     void clear(){
83         if (!this.isEmpty()) {
84             this.front = this.rear = -1;
85             this.size = 0;
86             System.out.println("Queue berhasil dikosongkan");
87         } else {
88             System.out.println("Queue masih kosong");
89         }
90     }
91 }
