  3 class DequeueResult {
  4     public boolean success;
  5     public Pembeli value;
  6     public DequeueResult(boolean s, Pembeli v){
  7         this.success = s;
  8         this.value = v;
  9     }
 10 }
 11 
 12 class Queue {
 13     Pembeli[] data;
 14     int front;
 15     int rear;
 16     int size;
 17     int max;
 18 
 19     Queue(int n){
 20         this.max = n;
 21         this.data = new Pembeli[this.max];
 22         this.size = 0;
 23         this.front = this.rear = -1;
 24     }
 25 
 26     boolean isEmpty(){
 27         return this.size == 0;
 28     }
 29 
 30     boolean isFull(){
 31         return this.size == this.max;
 32     }
 33 
 34     boolean enqueue(Pembeli dt){
 35         if (this.isFull()) {
 36             System.out.println("Queue sudah penuh");
 37             return false;
 38         }
 39         else {
 40             if (this.isEmpty())
 41                 this.front = this.rear = 0;
 42             else {
 43                 if (this.rear == this.max-1)
 44                     this.rear = 0;
 45                 else
 46                     this.rear++;
 47             }
 48             this.data[this.rear] = dt;
 49             this.size++;
 50             return true;
 51         }
 52     }
 53 
 54     DequeueResult dequeue(){
 55         Pembeli dt = new Pembeli();
 56         if (this.isEmpty()) {
 57             System.out.println("Queue kosong");
 58             return new DequeueResult(false, dt);
 59         }
 60         else {
 61             dt = this.data[this.front];
 62             this.size--;
 63             if (this.isEmpty())
 64                 this.front = this.rear = -1;
 65             else {
 66                 if (this.front == this.max-1)
 67                     this.front = 0;
 68                 else
 69                     this.front++;
 70             }
 71             return new DequeueResult(true, dt);
 72         }
 73     }
 74 
 75     void peek(){
 76         if (!this.isEmpty())
 77             System.out.println("Elemen terdepan: " +
 78                                 this.data[this.front].nama + " " +
 79                                 this.data[this.front].noHP);
 80         else
 81             System.out.println("Queue masih kosong");
 82     }
 83 
 84     void peekRear() {
 85         if (!this.isEmpty())
 86             System.out.println("Elemen terbelakang: " +
 87                                 this.data[this.front].nama + " " +
 88                                 this.data[this.front].noHP);
 89         else
 90             System.out.println("Queue kosong");
 91     }
 92 
 93     void peekPosition(String nama) {
 94         if (this.isEmpty()) {
 95             System.out.println("Queue kosong");
 96             return;
 97         }
 98         else {
 99             int i = front;
100             int posisi = 1;
101             while (i != rear) {
102                 if (this.data[i].nama.equals(nama)) {
103                     System.out.println(nama + " berada pada posisi " + posisi);
104                     return;
105                 }
106                 i = (i + 1) % this.max;
107                 posisi++;
108             }
109             if (this.data[i].nama.equals(nama)) {
110                 System.out.println(nama + " berada pada posisi " + posisi);
111                 return;
112             }
113         }
114         System.out.println(nama + " tidak berada dalam antrian");
115     }
116 
117     void daftarPembeli() {
118         if (this.isEmpty())
119             System.out.println("Queue masih kosong");
120         else {
121             int i = front;
122             while (i != rear) {
123                 System.out.println(this.data[this.front].nama + " " +
124                                    this.data[this.front].noHP);
125                 i = (i + 1) % this.max;
126             }
127                 System.out.println(this.data[this.front].nama + " " +
128                                    this.data[this.front].noHP);
129             System.out.println("Jumlah elemen = " + this.size);
130         }
131     }
132 
133     void print(){
134         if (this.isEmpty())
135             System.out.println("Queue masih kosong");
136         else {
137             int i = front;
138             while (i != rear) {
139                 System.out.println(this.data[this.front].nama + " " +
140                                    this.data[this.front].noHP);
141                 i = (i + 1) % this.max;
142             }
143                 System.out.println(this.data[this.front].nama + " " +
144                                    this.data[this.front].noHP);
145             System.out.println("Jumlah elemen = " + this.size);
146         }
147     }
