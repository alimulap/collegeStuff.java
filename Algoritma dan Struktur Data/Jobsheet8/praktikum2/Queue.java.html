  3 class DequeueResult {
  4     public boolean success;
  5     public Nasabah value;
  6     public DequeueResult(boolean s, Nasabah v){
  7         this.success = s;
  8         this.value = v;
  9     }
 10 }
 11 
 12 class Queue {
 13     Nasabah[] data;
 14     int front;
 15     int rear;
 16     int size;
 17     int max;
 18 
 19     Queue(int n){
 20         this.max = n;
 21         this.data = new Nasabah[this.max];
 22         this.size = 0;
 23         this.front = this.rear = -1;
 24     }
 25 
 26     boolean isEmpty(){
 27         return this.size == 0;
 28     }
 29 
 30     boolean isFull(){
 31         return this.size == this.max;
 32     }
 33 
 34     boolean enqueue(Nasabah dt){
 35         if (this.isFull()) {
 36             System.out.println("Queue sudah penuh");
 37             return false;
 38         }
 39         else {
 40             if (this.isEmpty())
 41                 this.front = this.rear = 0;
 42             else {
 43                 if (this.rear == this.max-1)
 44                     this.rear = 0;
 45                 else
 46                     this.rear++;
 47             }
 48             this.data[this.rear] = dt;
 49             this.size++;
 50             return true;
 51         }
 52     }
 53 
 54     DequeueResult dequeue(){
 55         Nasabah dt = new Nasabah();
 56         if (this.isEmpty()) {
 57             System.out.println("Queue kosong");
 58             return new DequeueResult(false, dt);
 59         }
 60         else {
 61             dt = this.data[this.front];
 62             this.size--;
 63             if (this.isEmpty())
 64                 this.front = this.rear = -1;
 65             else {
 66                 if (this.front == this.max-1)
 67                     this.front = 0;
 68                 else
 69                     this.front++;
 70             }
 71             return new DequeueResult(true, dt);
 72         }
 73     }
 74 
 75     void peek(){
 76         if (!this.isEmpty())
 77             System.out.println("Elemen terdepan: " +
 78                                 this.data[this.front].norek + " " +
 79                                 this.data[this.front].nama + " " +
 80                                 this.data[this.front].alamat + " " +
 81                                 this.data[this.front].umur + " " +
 82                                 this.data[this.front].saldo);
 83         else
 84             System.out.println("Queue masih kosong");
 85     }
 86 
 87     void peekRear() {
 88         if (!this.isEmpty())
 89             System.out.println("Elemen terbelakang: " +
 90                                 this.data[this.rear].norek + " " +
 91                                 this.data[this.rear].nama + " " +
 92                                 this.data[this.rear].alamat + " " +
 93                                 this.data[this.rear].umur + " " +
 94                                 this.data[this.rear].saldo);
 95         else
 96             System.out.println("Queue kosong");
 97     }
 98 
 99     void print(){
100         if (this.isEmpty())
101             System.out.println("Queue masih kosong");
102         else {
103             int i = front;
104             while (i != rear) {
105                 System.out.println(this.data[this.front].norek + " " +
106                                    this.data[this.front].nama + " " +
107                                    this.data[this.front].alamat + " " +
108                                    this.data[this.front].umur + " " +
109                                    this.data[this.front].saldo);
110                 i = (i + 1) % this.max;
111             }
112             System.out.println(this.data[i].norek + " " +
113                                this.data[i].nama + " " +
114                                this.data[i].alamat + " " +
115                                this.data[i].umur + " " +
116                                this.data[i].saldo);
117             System.out.println("Jumlah elemen = " + this.size);
118         }
119     }
120 
121     void clear(){
122         if (!this.isEmpty()) {
123             this.front = this.rear = -1;
124             this.size = 0;
125             System.out.println("Queue berhasil dikosongkan");
126         } else {
127             System.out.println("Queue masih kosong");
128         }
129     }
130 }
