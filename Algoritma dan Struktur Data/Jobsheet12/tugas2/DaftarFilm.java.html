  3 class Node {
  4     Film data;
  5     Node prev, next;
  6 
  7     Node(Node prev, Film data, Node next) {
  8         this.prev = prev;
  9         this.data = data;
 10         this.next = next;
 11     }
 12 }

 14 class DaftarFilm {
 15     private Node head;
 16     private int size;
 17 
 18     DaftarFilm() {
 19         this.head = null;
 20         this.size = 0;
 21     }
 22 
 23     public int size() {
 24         return this.size;
 25     }
 26 
 27     public boolean add(Film item, int index) {
 28         if (this.head == null)
 29             this.head = new Node(null, item, null);
 30         else if (index < 0 || index > this.size) {
 31             System.out.println("Index salah");
 32             return false;
 33         }
 34         else {
 35             Node current = this.head;
 36             int i = 0;
 37             while (i < index) {
 38                 current = current.next;
 39                 ++i;
 40             }
 41             if (current.prev == null) {
 42                 Node newNode = new Node(null, item, current);
 43                 current.prev = newNode;
 44                 this.head = newNode;
 45             } else {
 46                 Node newNode = new Node(current.prev, item, current);
 47                 newNode.prev = current.prev;
 48                 newNode.next = current;
 49                 current.prev.next = newNode;
 50                 current.prev = newNode;
 51             }
 52         }
 53         ++this.size;
 54         return true;
 55     }
 56 
 57     public boolean remove(int index) {
 58         if (this.head == null)
 59             System.out.println("Daftar kosong");
 60         else if (index < 0 || index >= this.size)
 61             System.out.println("Index salah");
 62         else {
 63             Node current = this.head;
 64             int i = 0;
 65             while (i < index) {
 66                 current = current.next;
 67                 ++i;
 68             }
 69             if (current.next == null)
 70                 current.prev.next = null;
 71             else if (current.prev == null) {
 72                 current = current.next;
 73                 current.prev = null;
 74                 this.head = current;
 75             } else {
 76                 current.prev.next = current.next;
 77                 current.next.prev = current.prev;
 78             }
 79             --this.size;
 80             return true;
 81         }
 82         return false;
 83     }

 85     public void print() {
 86         System.out.println("-----------------------");
 87         System.out.println("Daftar Film Layar Lebar");
 88         System.out.println("-----------------------");
 89         Node current = this.head;
 90         while (current != null) {
 91             current.data.print();
 92             current = current.next;
 93         }
 94         System.out.println("\n");
 95     }

 97     public void cariFillm(int id) {
 98         Node current = this.head;
 99         int i = 0;
100         while (current != null) {
101             if (current.data.id == id) {
102                 System.out.println("Data dengan ID Film " + id + " berada pada node ke-" + (i+1) + "\nIdentitas:\n");
103                 current.data.print();
104                 return;
105             }
106             current = current.next;
107             ++i;
108         }
109         System.out.println("Data dengan ID Film " + id + " tidak ditemukan didalam daftar film");
110     }

112     public void sortByRating() {
113         Node current = head;
114         while (current != null) {
115             Node temp = current.next;
116             while (temp != null) {
117                 if (temp.data.id > current.data.id) {
118                     Film tempData = current.data;
119                     current.data = temp.data;
120                     temp.data = tempData;
121                 }
122                 temp = temp.next;
123             }
124             current = current.next;
125         }
126     }
