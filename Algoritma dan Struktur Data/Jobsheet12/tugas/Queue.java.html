 3 class Node {
 4     PenerimaVaksin data;
 5     Node next, prev;
 6 
 7     Node(Node prev, PenerimaVaksin data, Node next) {
 8         this.data = data;
 9         this.prev = prev;
10         this.next = next;
11     }
12 }
13 
14 class Queue {
15     Node head, tail;
16     int size;
17 
18     Queue() {
19         this.head = null;
20         this.tail = null;
21         this.size = 0;
22     }
23 
24     public void enqueue(PenerimaVaksin input) {
25         if (this.head == null) {
26             this.head = new Node(null, input, null);
27             this.tail = this.head;
28         } else {
29             Node newNode = new Node(this.tail, input, null);
30             this.tail.next = newNode;
31             this.tail = newNode;
32         }
33         ++this.size;
34     }
35 
36     public void dequeue() {
37         if (this.head == null)
38             System.out.println("Antrian kosong.");
39         else {
40             System.out.println(this.head.data.nama + " telah selesai divaksinasi.");
41             this.head.next.prev = null;
42             this.head = this.head.next;
43             --this.size;
44             this.print();
45         }
46     }
47 
48     public void print() {
49         System.out.println("+++++++++++++++++++++++");
50         System.out.println("Daftar Pengantri Vaksin");
51         System.out.println("+++++++++++++++++++++++");
52         System.out.println("|No.\t|Nama.\t|");
53         Node current = this.head;
54         while (current != this.tail) {
55             System.out.println("|" + current.data.nomor + "\t|" + current.data.nama + "\t|");
56             current = current.next;
57         }
58         System.out.println("|" + current.data.nomor + "\t|" + current.data.nama + "\t|");
59         System.out.println("Sisa Antrian: " + this.size);
60     }
61 }
