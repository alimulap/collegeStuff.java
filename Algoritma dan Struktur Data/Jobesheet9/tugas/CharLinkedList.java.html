 20 class Node {
 21     char data;
 22     Node next;
 23 
 24     Node(char nilai, Node berikutnya) {
 25         this.data = nilai;
 26         this.next = berikutnya;
 27     }
 28 }
 29 
 30 class SingleLinkedList {
 31     Node head;
 32     Node tail;
 33 
 34     public boolean isEmpty() {
 35         return this.head == null;
 36     }
 37 
 38     public void print() {
 39         if (!this.isEmpty()) {
 40             Node tmp = head;
 41             System.out.print("Isi Linked List:\t");
 42             while (tmp != null) {
 43                 System.out.print(tmp.data + "\t");
 44                 tmp = tmp.next;
 45             }
 46             System.out.println();
 47         } else {
 48             System.out.println("Linked List kosong");
 49         }
 50     }
 51 
 52     public void addFirst(char input) {
 53         Node ndInput = new Node(input, null);
 54         if (this.isEmpty()) {
 55             this.head = ndInput;
 56             this.tail = ndInput;
 57         } else {
 58             ndInput.next = this.head;
 59             this.head = ndInput;
 60         }
 61     }
 62 
 63     public void addLast(char input) {
 64         Node ndInput = new Node(input, null);
 65         if (this.isEmpty()) {
 66             this.head = ndInput;
 67             this.tail = ndInput;
 68         } else {
 69             this.tail.next = ndInput;
 70             this.tail = ndInput;
 71         }
 72     }
 73 
 74     public void insertAfter(char key, char input) {
 75         Node ndInput = new Node(input, null);
 76         Node temp = head;
 77         do {
 78             if (temp.data == key) {
 79                 ndInput.next = temp.next;
 80                 temp.next = ndInput;
 81                 if (ndInput.next == null) this.tail = ndInput;
 82                 break;
 83             }
 84             temp = temp.next;
 85         } while (temp != null);
 86     }
 87 
 88     public void insertAt(int index, char input) {
 89         if (index < 0) {
 90             System.out.println("index salah");
 91         } else if (index == 0) {
 92             this.addFirst(input);
 93         } else {
 94             Node temp = this.head;
 95             for (int i = 0; i < index -1; ++i)
 96                 temp = temp.next;
 97             temp.next = new Node(input, temp.next);
 98             if (temp.next == null) this.tail = temp.next;
 99         }
100     }
