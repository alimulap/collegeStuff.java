 11     public void print() {
 12         if (!this.isEmpty()) {
 13             Node tmp = head;
 14             System.out.print("Isi Linked List:\t");
 15             while (tmp != null) {
 16                 System.out.print(tmp.data + "\t");
 17                 tmp = tmp.next;

 21             System.out.println("Linked List kosong");
 22         }

 25     public void addFirst(int input) {
 26         Node ndInput = new Node(input, null);
 27         if (this.isEmpty()) {
 28             this.head = ndInput;
 29             this.tail = ndInput;
 30         } else {
 31             ndInput.next = this.head;
 32             this.head = ndInput;
 33         }
 34     }

 36     public void addLast(int input) {
 37         Node ndInput = new Node(input, null);
 38         if (this.isEmpty()) {
 39             this.head = ndInput;
 40             this.tail = ndInput;

 42             this.tail.next = ndInput;

 47     public void insertAfter(int key, int input) {
 48         Node ndInput = new Node(input, null);
 49         Node temp = head;
 50         do {
 51             if (temp.data == key) {
 52                 ndInput.next = temp.next;
 53                 temp.next = ndInput;
 54                 if (ndInput.next == null) this.tail = ndInput;
 55                 break;

 57             temp = temp.next;

 61     public void insertBefore(int key, int input) {
 62         Node ndInput = new Node(input, null);
 63         Node temp = head;
 64         if (temp.data == key) {
 65             ndInput.next = this.head;
 66             this.head = ndInput;
 67         }
 68         do {
 69             if (temp.next.data == key) {
 70                 ndInput.next = temp.next;
 71                 temp.next = ndInput;
 72                 break;
 73             }
 74             temp = temp.next;
 75         } while (temp != null);
 76     }

 78     public void insertAt(int index, int input) {
 79         if (index < 0) {
 80             System.out.println("index salah");
 81         } else if (index == 0) {
 82             this.addFirst(input);
 83         } else {
 84             Node temp = this.head;
 85             for (int i = 0; i < index -1; ++i)
 86                 temp = temp.next;
 87             temp.next = new Node(input, temp.next);
 88             if (temp.next == null) this.tail = temp.next;
 89         }

 99     public int indexOf(int key) {
100         Node tmp = this.head;
101         int index = 0;
102         while (tmp != null && tmp.data != key) {
103             tmp = tmp.next;
104             ++index;
105         }
106         if (tmp == null)
107             return -1;
108         else
109             return index;
110     }

116             this.head = this.tail = null;

122         if (this.isEmpty())
123             System.out.println("LinkedList masih kosong, tidak dapat menghapus elemen");
124         else if (this.head == this.tail)
125             this.head = this.tail = null;

127             Node temp = this.head;

135     public void remove(int key) {
136         if (this.isEmpty())
137             System.out.println("LinkedList masih kosong, tidak dapat menghapus elemen");
138         else {
139             Node temp = this.head;
140             while (temp != null) {
141                 if ((temp.data == key) && (temp == head)) {
142                     this.removeFirst();
143                     break;
144                 } else if (temp.next.data == key) {
145                     temp.next = temp.next.next;
146                     if (temp.next == null)
147                         this.tail = temp;
148                     break;
149                 }
150                 temp = temp.next;
151             }
152         }
153     }

155     public void removeAt(int index) {
156         if (index == 0)
157             this.removeFirst();
158         else {
159             Node temp = this.head;
160             for (int i = 0; i < index - 1; ++i)
161                 temp = temp.next;
162             temp.next = temp.next.next;
