 18             Node current = this.root;
 19             while (true) {

144     Node getSuccessor(Node del) {
145         Node successor = del.right;
146         Node successorParent = del;
147         while (successor.left != null) {
148             successorParent = successor;
149             successor = successor.left;
150         }
151         if (successor != del.right) {
152             successorParent.left = successor.right;
153             successor.right = del.right;
154         }
155         return successor;

158     void delete(int data) {
159         if (this.isEmpty()) {
160             System.out.println("Tree is empty");
161             return;
162         }
163 
164         Node parent = this.root;
165         Node current = this.root;
166         boolean isLeftChild = false;
167         while (current != null) {
168             if (current.data == data)
169                 break;
170             else if (data < current.data) {
171                 parent = current;
172                 current = current.left;
173                 isLeftChild = true;
174             } else if (data > current.data) {
175                 parent = current;
176                 current = current.right;
177                 isLeftChild = false;
178             }
179         }
180 
181         if (current == null) {
182             System.out.println("Couldn't find data!");
183             return;
184         } else {
185             if (current.left == null && current.right == null) {
186                 if (current == this.root)
187                     this.root = null;
188                 else {
189                     if (isLeftChild)
190                         parent.left = null;
191                     else
192                         parent.right = null;
193                 }
194             } else if (current.left == null) {
195                 if (current == this.root)
196                     this.root = current.right;
197                 else {
198                     if (isLeftChild)
199                         parent.left = current.right;
200                     else
201                         parent.right = current.right;
202                 }
203             } else if (current.right == null) {
204                 if (current == this.root)
205                     this.root = current.left;
206                 else {
207                     if (isLeftChild)
208                         parent.left = current.left;
209                     else
210                         parent.right = current.left;
211                 }
212             } else {
213                 Node successor = this.getSuccessor(current);
214                 if (current == this.root)
215                     this.root = successor;
216                 else {
217                     if (isLeftChild)
218                         parent.left = successor;
219                     else
220                         parent.right = successor;
221                     successor.left = current.left;
222                 }
223             }
224         }
